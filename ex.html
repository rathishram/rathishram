// Function to make a GET request using Axios async function getTodos() { try { const response = await axios.get('https://jsonplaceholder.typicode.com/todos/1'); showOutput(response); } catch (error) { console.error(`Error: ${error.message}`); } } // Function to show output in the browser function showOutput(res) { document.getElementById('res').innerHTML = `
Status: ${res.status}
Headers 
${JSON.stringify(res.headers, null, 2)}
Data 
${JSON.stringify(res.data, null, 2)}
Config 
${JSON.stringify(res.config, null, 2)}
`; } // Event listener for the 'GET' button document.getElementById('get').addEventListener('click', getTodos);
